/* ============================================================================
   TEST GEMINI LIVE — Form Softr (Vibe Coding block) — brand Sharazad
   Tabella: "Test Gemini Live" (db Tacita 2.0). Campi: Nome, Cognome, Ruolo, Azienda.
   Flusso: crea record -> POST /api/auth-request (valida @sharazad.com, invia mail
   col link ?recordId) -> redirect a pagina "controlla email". Solo @sharazad.com.
   Email NON salvata su Softr (solo per invio link).

   NOTA campi: q.select usa i NOMI campo. Se Vibe Coding richiede gli ID interni e
   il create fallisce, sostituisci i nomi con gli ID (Sources -> Content / lista campi).
   ============================================================================ */
import { Button } from "@/components/ui/button";
import { Input } from "@/components/ui/input";
import { Label } from "@/components/ui/label";
import { useTextSetting, useLongTextSetting } from "@/lib/editable-settings";
import { useState } from "react";
import { toast } from "sonner";
import { DynamicIcon } from "@/components/dynamic-icon";
import { useRecordCreate, q } from "@/lib/datasource";

// Mappatura campi tabella "Test Gemini Live". I valori sono i NOMI campo Softr.
// Se Vibe Coding richiede gli ID interni e il create fallisce, sostituisci i nomi
// con gli ID (Sources → tab Content / lista campi).
const fields = q.select({
  nome: "Nome",
  cognome: "Cognome",
  ruolo: "Ruolo",
  azienda: "Azienda",
});

// Page 2 "controlla email" (pubblica) — dopo il submit reindirizziamo qui.
const CHECK_EMAIL_URL = "https://yajaira65701.softr.app/daticustom-test-gemini";
// Endpoint server: valida dominio @sharazad.com e invia la mail col link intervista.
const AUTH_REQUEST_URL = "https://gemini-live-interview.onrender.com/api/auth-request";

export default function Block() {
  const title = useTextSetting({
    name: "title",
    label: "Form Title",
    initialValue: "Test intervista AI",
  });
  const welcomeText = useLongTextSetting({
    name: "welcome-text",
    label: "Welcome Text",
    initialValue: "Compila i tuoi dati: servono a personalizzare l'intervistatore AI (apertura e domande adattate al tuo ruolo e alla tua azienda).",
  });
  const submitButtonText = useTextSetting({
    name: "submit-button-text",
    label: "Submit Button Text",
    initialValue: "Avvia",
  });

  const [formData, setFormData] = useState({ nome: "", cognome: "", ruolo: "", azienda: "" });
  // Email NON salvata su Softr: serve solo a inviare il link via mail (solo @sharazad.com).
  const [emailAccesso, setEmailAccesso] = useState("");

  const createRecord = useRecordCreate({
    fields,
    onSuccess: async (record: any) => {
      const recordId = record?.id || "";
      const nome = formData.nome.trim();
      const cognome = formData.cognome.trim();
      const email = emailAccesso.trim().toLowerCase();
      toast.success("Dati salvati!");
      // Richiesta invio link via email (verifica dominio lato server)
      try {
        await fetch(AUTH_REQUEST_URL, {
          method: "POST",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify({ nome, cognome, email, recordId }),
        });
      } catch (err) { console.error("[authRequest]", err); }
      setFormData({ nome: "", cognome: "", ruolo: "", azienda: "" });
      setEmailAccesso("");
      window.location.href = CHECK_EMAIL_URL;
    },
    onError: (error: any) => toast.error(`Errore durante l'invio: ${error.message}`),
  });

  const handleChange = (field: string, value: string) =>
    setFormData((prev) => ({ ...prev, [field]: value }));

  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault();
    createRecord.mutate(formData);
  };

  const inputStyle = {
    backgroundColor: "#f4f4fb", border: "1px solid rgba(0,0,0,0.08)",
    borderRadius: "12px", color: "#151927", fontWeight: 400,
  } as React.CSSProperties;

  return (
    <>
      <style>{`
        @import url('https://fonts.googleapis.com/css2?family=DM+Sans:wght@300;400;500;600;700&display=swap');
        body { font-family: 'DM Sans', sans-serif; }
      `}</style>

      <div className="min-h-screen bg-white relative overflow-hidden" style={{ fontFamily: "'DM Sans', sans-serif" }}>
        <div className="absolute inset-0 pointer-events-none overflow-hidden">
          <div className="absolute" style={{ bottom: "-200px", left: "-200px", width: "600px", height: "600px", background: "radial-gradient(circle, rgba(84,81,208,0.20) 0%, transparent 70%)", filter: "blur(60px)", zIndex: 0 }} />
          <div className="absolute" style={{ top: "-150px", right: "-150px", width: "500px", height: "500px", background: "radial-gradient(circle, rgba(84,81,208,0.12) 0%, transparent 70%)", filter: "blur(50px)", zIndex: 0 }} />
        </div>

        <div className="container relative z-10 py-16">
          <div className="content max-w-3xl mx-auto">
            <div className="bg-white border rounded-2xl p-8 lg:p-10" style={{ borderColor: "rgba(0,0,0,0.08)", boxShadow: "0 4px 16px rgba(0,0,0,0.06)" }}>
              <img src="https://gemini-live-interview.onrender.com/sharazad-logo.png" alt="Sharazad" style={{ height: "38px", width: "auto", marginBottom: "20px" }} />
              <div className="inline-flex items-center gap-2 px-5 py-2.5 rounded-full text-sm mb-6" style={{ backgroundColor: "rgba(84,81,208,0.1)", color: "#4340a8", fontWeight: 600 }}>
                <DynamicIcon name="sparkles" className="w-5 h-5" />
                Intervista AI · Gemini Live
              </div>
              <h1 className="text-3xl lg:text-4xl leading-tight mb-3" style={{ color: "#151927", fontWeight: 700 }}>{title}</h1>
              <p className="text-base leading-relaxed mb-8" style={{ color: "#6b7280", fontWeight: 400 }}>{welcomeText}</p>

              <form onSubmit={handleSubmit} className="space-y-6">
                <div className="grid md:grid-cols-2 gap-5">
                  <div className="space-y-2">
                    <Label htmlFor="nome" style={{ color: "#151927", fontWeight: 500 }}>Nome <span style={{ color: "#dc2626" }}>*</span></Label>
                    <Input id="nome" value={formData.nome} onChange={(e) => handleChange("nome", e.target.value)} required className="h-14 text-lg" placeholder="Il tuo nome" style={inputStyle} />
                  </div>
                  <div className="space-y-2">
                    <Label htmlFor="cognome" style={{ color: "#151927", fontWeight: 500 }}>Cognome <span style={{ color: "#dc2626" }}>*</span></Label>
                    <Input id="cognome" value={formData.cognome} onChange={(e) => handleChange("cognome", e.target.value)} required className="h-14 text-lg" placeholder="Il tuo cognome" style={inputStyle} />
                  </div>
                </div>
                <div className="space-y-2">
                  <Label htmlFor="emailAccesso" style={{ color: "#151927", fontWeight: 500 }}>Email aziendale <span style={{ color: "#dc2626" }}>*</span></Label>
                  <Input id="emailAccesso" type="email" value={emailAccesso} onChange={(e) => setEmailAccesso(e.target.value)} required className="h-14 text-lg" placeholder="nome@sharazad.com" style={inputStyle} />
                  <p className="text-sm" style={{ color: "#6b7280" }}>Il link personale per l'intervista verrà inviato a questa email.</p>
                </div>
                <div className="space-y-2">
                  <Label htmlFor="ruolo" style={{ color: "#151927", fontWeight: 500 }}>Ruolo <span style={{ color: "#dc2626" }}>*</span></Label>
                  <Input id="ruolo" value={formData.ruolo} onChange={(e) => handleChange("ruolo", e.target.value)} required className="h-14 text-lg" placeholder="Es. Responsabile IT, Designer, Manager..." style={inputStyle} />
                </div>
                <div className="space-y-2">
                  <Label htmlFor="azienda" style={{ color: "#151927", fontWeight: 500 }}>Azienda <span style={{ color: "#dc2626" }}>*</span></Label>
                  <Input id="azienda" value={formData.azienda} onChange={(e) => handleChange("azienda", e.target.value)} required className="h-14 text-lg" placeholder="Nome azienda" style={inputStyle} />
                </div>

                <Button
                  type="submit"
                  disabled={createRecord.status === "pending" || !createRecord.enabled}
                  className="w-full h-14 text-lg mt-4"
                  style={{ backgroundColor: "#5451d0", color: "white", borderRadius: "12px", fontWeight: 600, boxShadow: "0 2px 8px rgba(84,81,208,0.25)" }}
                >
                  {createRecord.status === "pending" ? (
                    <span className="flex items-center gap-2"><DynamicIcon name="loader-2" className="w-5 h-5 animate-spin" /> Invio…</span>
                  ) : (
                    <span className="flex items-center gap-2">{submitButtonText} <DynamicIcon name="arrow-right" className="w-5 h-5" /></span>
                  )}
                </Button>
              </form>
            </div>
            <p style={{ textAlign: "center", fontSize: "11px", letterSpacing: ".15em", color: "#b0b8c4", marginTop: "20px", textTransform: "uppercase" }}>
              Test interno · tecnologia Sharazad · voce elaborata in UE
            </p>
          </div>
        </div>
      </div>
    </>
  );
}
